package Entity;

import java.util.Calendar;

/**
 *
 * @author idea
 */
public class Yorum {
    
    private int yorum_id;
    private int film_id;
    private int user_id;
    private String icerik;
    private String yorum_basligi;
    private Calendar yorumolusturma_tarihi;

    public Yorum() {
    }

    public Yorum(int yorum_id, int film_id, int user_id, String icerik, String yorum_basligi, Calendar yorumolusturma_tarihi) {
        this.yorum_id = yorum_id;
        this.film_id = film_id;
        this.user_id = user_id;
        this.icerik = icerik;
        this.yorum_basligi = yorum_basligi;
        this.yorumolusturma_tarihi = yorumolusturma_tarihi;
    }

    public int getYorum_id() {
        return yorum_id;
    }

    public void setYorum_id(int yorum_id) {
        this.yorum_id = yorum_id;
    }

    public int getFilm_id() {
        return film_id;
    }

    public void setFilm_id(int film_id) {
        this.film_id = film_id;
    }

    public int getUser_id() {
        return user_id;
    }

    public void setUser_id(int user_id) {
        this.user_id = user_id;
    }

    public String getIcerik() {
        return icerik;
    }

    public void setIcerik(String icerik) {
        this.icerik = icerik;
    }

    public String getYorum_basligi() {
        return yorum_basligi;
    }

    public void setYorum_basligi(String yorum_basligi) {
        this.yorum_basligi = yorum_basligi;
    }

    public Calendar getYorumolusturma_tarihi() {
        return yorumolusturma_tarihi;
    }

    public void setYorumolusturma_tarihi(Calendar yorumolusturma_tarihi) {
        this.yorumolusturma_tarihi = yorumolusturma_tarihi;
    }
      
      
    
}
