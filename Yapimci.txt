package Entity;

/**
 *
 * @author idea
 */
public class Yapimci {
    
    private int yapimci_id;
    private String yapimci_adi;
    private int kurulus_tarihi;
    private String mail;

    public Yapimci() {
    }

    public Yapimci(int yapimci_id, String yapimci_adi, int kurulus_tarihi, String mail) {
        this.yapimci_id = yapimci_id;
        this.yapimci_adi = yapimci_adi;
        this.kurulus_tarihi = kurulus_tarihi;
        this.mail = mail;
    }

    public int getYapimci_id() {
        return yapimci_id;
    }

    public void setYapimci_id(int yapimci_id) {
        this.yapimci_id = yapimci_id;
    }

    public String getYapimci_adi() {
        return yapimci_adi;
    }

    public void setYapimci_adi(String yapimci_adi) {
        this.yapimci_adi = yapimci_adi;
    }

    public int getKurulus_tarihi() {
        return kurulus_tarihi;
    }

    public void setKurulus_tarihi(int kurulus_tarihi) {
        this.kurulus_tarihi = kurulus_tarihi;
    }

    public String getMail() {
        return mail;
    }

    public void setMail(String mail) {
        this.mail = mail;
    }
    
}
