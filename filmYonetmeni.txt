package Entity;

 */
public class filmYonetmeni {
    
    private int filmyonetmeni_id;
    private String yonetmen_adi;
    private String yonetmen_soyadi;
    private String yonetmen_cinsiyeti;

    public filmYonetmeni() {
    }

    public filmYonetmeni(int filmyonetmeni_id, String yonetmen_adi, String yonetmen_soyadi, String yonetmen_cinsiyeti) {
        this.filmyonetmeni_id = filmyonetmeni_id;
        this.yonetmen_adi = yonetmen_adi;
        this.yonetmen_soyadi = yonetmen_soyadi;
        this.yonetmen_cinsiyeti = yonetmen_cinsiyeti;
    }

    public int getFilmyonetmeni_id() {
        return filmyonetmeni_id;
    }

    public void setFilmyonetmeni_id(int filmyonetmeni_id) {
        this.filmyonetmeni_id = filmyonetmeni_id;
    }

    public String getYonetmen_adi() {
        return yonetmen_adi;
    }

    public void setYonetmen_adi(String yonetmen_adi) {
        this.yonetmen_adi = yonetmen_adi;
    }

    public String getYonetmen_soyadi() {
        return yonetmen_soyadi;
    }

    public void setYonetmen_soyadi(String yonetmen_soyadi) {
        this.yonetmen_soyadi = yonetmen_soyadi;
    }

    public String getYonetmen_cinsiyeti() {
        return yonetmen_cinsiyeti;
    }

    public void setYonetmen_cinsiyeti(String yonetmen_cinsiyeti) {
        this.yonetmen_cinsiyeti = yonetmen_cinsiyeti;
    }
    
}
