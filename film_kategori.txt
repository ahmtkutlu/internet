package Entity;

/**
 *
 * @author idea
 */
public class film_kategori {
    
    private int film_kategori_id;
    private int film_id;
    private int kategori_id;

    public film_kategori() {
    }

    public film_kategori(int film_kategori_id, int film_id, int kategori_id) {
        this.film_kategori_id = film_kategori_id;
        this.film_id = film_id;
        this.kategori_id = kategori_id;
    }

    public int getFilm_kategori_id() {
        return film_kategori_id;
    }

    public void setFilm_kategori_id(int film_kategori_id) {
        this.film_kategori_id = film_kategori_id;
    }

    public int getFilm_id() {
        return film_id;
    }

    public void setFilm_id(int film_id) {
        this.film_id = film_id;
    }

    public int getKategori_id() {
        return kategori_id;
    }

    public void setKategori_id(int kategori_id) {
        this.kategori_id = kategori_id;
    }
    
}
