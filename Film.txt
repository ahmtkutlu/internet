package Entity;

/**
 *
 * @author idea
 */
public class Film {
    private int film_id;
    private String film_ad;
    private int ımdb_puani;
    private int sure;
    private int butce;
    private int yil;
    private int filmyonetmeni_id;
    private int isikyonetmeni_id;
    private int sesyonetmeni_id;
    private int yapimci_id;
    private String film_icerik;

    public Film() {
    }

    public Film(int film_id, String film_ad, int ımdb_puani, int sure, int butce, int yil, int filmyonetmeni_id, int isikyonetmeni_id, int sesyonetmeni_id, int yapimci_id, String film_icerik) {
        this.film_id = film_id;
        this.film_ad = film_ad;
        this.ımdb_puani = ımdb_puani;
        this.sure = sure;
        this.butce = butce;
        this.yil = yil;
        this.filmyonetmeni_id = filmyonetmeni_id;
        this.isikyonetmeni_id = isikyonetmeni_id;
        this.sesyonetmeni_id = sesyonetmeni_id;
        this.yapimci_id = yapimci_id;
        this.film_icerik = film_icerik;
    }

    public int getFilm_id() {
        return film_id;
    }

    public void setFilm_id(int film_id) {
        this.film_id = film_id;
    }

    public String getFilm_ad() {
        return film_ad;
    }

    public void setFilm_ad(String film_ad) {
        this.film_ad = film_ad;
    }

    public int getImdb_puani() {
        return ımdb_puani;
    }

    public void setImdb_puani(int ımdb_puani) {
        this.ımdb_puani = ımdb_puani;
    }

    public int getSure() {
        return sure;
    }

    public void setSure(int sure) {
        this.sure = sure;
    }

    public int getButce() {
        return butce;
    }

    public void setButce(int butce) {
        this.butce = butce;
    }

    public int getYil() {
        return yil;
    }

    public void setYil(int yil) {
        this.yil = yil;
    }

    public int getFilmyonetmeni_id() {
        return filmyonetmeni_id;
    }

    public void setFilmyonetmeni_id(int filmyonetmeni_id) {
        this.filmyonetmeni_id = filmyonetmeni_id;
    }

    public int getIsikyonetmeni_id() {
        return isikyonetmeni_id;
    }

    public void setIsikyonetmeni_id(int isikyonetmeni_id) {
        this.isikyonetmeni_id = isikyonetmeni_id;
    }

    public int getSesyonetmeni_id() {
        return sesyonetmeni_id;
    }

    public void setSesyonetmeni_id(int sesyonetmeni_id) {
        this.sesyonetmeni_id = sesyonetmeni_id;
    }

    public int getYapimci_id() {
        return yapimci_id;
    }

    public void setYapimci_id(int yapimci_id) {
        this.yapimci_id = yapimci_id;
    }

    public String getFilm_icerik() {
        return film_icerik;
    }

    public void setFilm_icerik(String film_icerik) {
        this.film_icerik = film_icerik;
    }
            
    
}
